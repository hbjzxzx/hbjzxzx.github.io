Divide And Conquer

1. Divide the problem into one or some subproblems

2. Conquer each subproblem recursively

3. Combine solutions

Ex1 Merge Sort

Divide: trivial, just divice it into half!

Conquer: recursively solves each array

Combine: Merge the tow ordered array

Running Time: $T(n)=2T(\frac{n}{2}) + n$ that is $O(n \lg n)$

def merge_sort(array: List[int]) -> List[int]:
    a_len = len(array)
    if a_len <= 1:
        return array
    fst, snd = array[:a_len//2], array[a_len//2:]
    fst = merge_sort(fst)
    snd = merge_sort(snd)
    return merge(fst, snd)

def merge(xs1: List[int], xs2: List[int]) -> List[int]:
    result = []
    while xs1 and xs2:
        if xs1[0] > xs2[0]:
            result.append(xs2[0])
            xs2 = xs2[1:]
        elif xs1[0] <= xs2[0]:
            result.append(xs1[0])
            xs1 = xs1[1:]
    if xs1:
        result.extend(xs1) 
    if xs2:
        result.extend(xs2)
    return result

l, r, max_len, arr_cnt = 0, 10000, 10, 10
rand_list = map(lambda alen: [randint(l, r) for _ in range(alen)], 
                    [randint(0, max_len) for _ in range(arr_cnt) ]) 
for arr in tqdm(rand_list):
    try:
        assert merge_sort(arr) == sorted(arr)
    except AssertionError:
        print(f"assert failed, original: {arr}, std: {sorted(arr)}, my:{merge_sort(arr)}")

Ex2 Binary Search

Divide: compare x with middle elem in array

Conquer: recurse in one subarray

Combine: trivial, nothing to do!

Running Time: $T(n)=T(\frac{n}{2})+O(1)$ that is $O(\lg n)$

def binary_search(arr: List[int], lo:int, hi:int, num: int) -> int:
    if lo == hi:
        return -1
    mid = (lo + hi) // 2
    if arr[mid] > num:
        return binary_search(arr, lo, mid, num)
    elif arr[mid] < num:
        return binary_search(arr, mid+1, hi, num)
    else:
        return mid

l, r, ll = 0, 10000000, 500
rand_list = sorted([randint(l, r) for _ in range(ll)])
for _ in tqdm(range(100000)):
    sel = randint(0, ll-1)
    try:
        find_index = binary_search(rand_list, 0, len(rand_list), rand_list[sel])
        assert rand_list[find_index] == rand_list[sel]
    except:
        print(f"epect :{sel}, return {find_index}, input: {rand_list}")
        break

Ex3 Powwering a number: given number x, integer $n \ge 0$, compute $x^n$

Divide:

$$ sign(x) = \left{ \begin{aligned} x^{\frac{n}{2}} \cdot x^{\frac{n}{2}} \quad even \

x^{\frac{n-1}{2}} \cdot x^{\frac{n-1}{2}} \cdot x \quad odd \ \end{aligned} \right. $$

def power(x: float, y: float) -> int :
    if y == 1:
        return x
    if y % 2 == 1:
        return power(x, (y-1)//2) * x * power(x, (y-1)//2)
    else:
        return power(x, y//2) * power(x, y//2)
cnt = randint(0, 1000)
base_power_list = [(randint(1, 20), randint(1, 100)) for _ in range(cnt)]
assert all(
    map(lambda b, p: (b**p - power(b, p))<= 0.001, *zip(*base_power_list))
)
