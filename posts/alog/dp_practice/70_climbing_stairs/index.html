Climbing Stairs

You are climbing a staircase. It takes n steps to reach the top.Each time you can either climb 1 or 2 steps. In how many distinct ways can you climb to the top?

Original problem Leetcode 70

Recursive code

# use the recursice f(n) = f(n-1) + f(n-2) for all n > 2
class Solution:
	def climbStairs(self, n: int) -> int:
  	if n == 1:
    	return 1
  	if n == 2:
    	return 2
  	return f(n-1) + f(n-2)

# you can also use cache to boost up
from functools import cache

Iterate

from itertools import dropwhile
from typing import Iterable
class Solution:
    def climbStairs(self, n: int) -> int:
        infinite_series = enumerate(Solution.series())
        infinite_series = dropwhile(lambda pair: pair[0] < n-1, infinite_series)
        return next(infinite_series)[1]
    
    @staticmethod
    def series() -> Iterable[int]:
        s1, s2 = 1, 2
        yield s1
        yield s2
        while True:
            tmp = s2
            s2 = s1 + s2
            s1 = tmp
            yield s2

Matrix Multiply

Notice that the iterating states of f(n) = f(n-1) + f(n-2) can be described by matrix multiply $$ \begin{align} \begin{bmatrix}f(n)\ f(n-1)\end{bmatrix} = \begin{bmatrix}1& 1\ 1 & 0\\end{bmatrix} * \begin{bmatrix}f(n-1) \ f(n-2)\end{bmatrix} \ \begin{bmatrix}f(n) \ f(n-1)\end{bmatrix} =\begin{bmatrix}1 & 1\ 1 & 0\end{bmatrix}^{n-2} * \begin{bmatrix} f(2) \ f(1) \end{bmatrix} \end{align} $$

from itertools import repeat
from typing import Iterable
from copy import deepcopy
import numpy
M = Iterable[Iterable[int]]
class Solution:
    def climbStairs(self, n: int) -> int:
        transform_matrix = ((1, 1), (1, 0))
        init_vector = ((2,), (1,))
        get_matrix_r1_c1 = lambda m: next(next(m))
        if n == 1:
           return 1
        if n == 2:
            return 2
        return get_matrix_r1_c1(Solution.matrix_mul(
           Solution.matrix2_power(transform_matrix, n-2),
           init_vector
            ))
       

    @staticmethod
    def matrix2_power(matrix: M, pow: int) -> M:
        if pow == 0:
            return ((1, 0), (0, 1))
        # power a row major matrix
        if pow == 1:
            return matrix
        if pow == 2:
            return Solution.matrix_mul(matrix, matrix)
        if pow % 2 == 0:
            return Solution.matrix2_power(
                Solution.matrix2_power(matrix, pow//2),
                2
            )
        else:
            return Solution.matrix_mul(
                matrix,
                Solution.matrix2_power(matrix, pow-1)
            )
            
    
    @staticmethod
    def matrix_mul(ma: M, mb: M) -> M:
        if ma == mb:
            mb = deepcopy(ma)
        mb_col_major = list(zip(*mb))
        col_size = len(mb_col_major) 
        sum_vecs = lambda vecs: map(sum, zip(*vecs))
        vec_multi_scalar = lambda vec, scalar: map(lambda raw: raw*scalar, vec)
        matrix_multi_vec = lambda matrix, vec: sum_vecs(map(vec_multi_scalar, matrix, vec))
        
        return map(lambda matrix, col_vec: matrix_multi_vec(matrix, col_vec), 
                        [deepcopy(m) for m in repeat(ma, col_size)], 
                        mb_col_major)
